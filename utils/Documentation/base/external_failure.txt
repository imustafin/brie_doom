class 
	EXTERNAL_FAILURE

inherit
	EIFFEL_RUNTIME_EXCEPTION

create 
	default_create

feature -- Access

	frozen code: INTEGER_32
			-- Exception code
		do
			Result := {EXCEP_CONST}.external_exception
		end

	Tag: IMMUTABLE_STRING_32
			-- A short message describing what current exception is
		once
			create Result.make_from_string_8 ("External event.")
		end
	
end -- class EXTERNAL_FAILURE
