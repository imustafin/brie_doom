class 
	MISMATCH_FAILURE

inherit
	DATA_EXCEPTION

create 
	default_create

feature -- Access

	frozen code: INTEGER_32
			-- Exception code
		do
			Result := {EXCEP_CONST}.retrieve_exception
		end

	Tag: IMMUTABLE_STRING_32
			-- A short message describing what current exception is
		once
			create Result.make_from_string_8 ("Mismatch failed.")
		end
	
end -- class MISMATCH_FAILURE
