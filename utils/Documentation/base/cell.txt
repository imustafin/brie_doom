class 
	CELL [G]

create 
	put

feature -- Access

	item: G
			-- Content of cell.
	
feature -- Element change

	put (v: like item)
			-- Make `v` the cell's item.
			-- Was declared in CELL as synonym of replace.
		do
			item := v
		ensure
			item_inserted: item = v
		end

	replace (v: like item)
			-- Make `v` the cell's item.
			-- Was declared in CELL as synonym of put.
		do
			item := v
		ensure
			item_inserted: item = v
		end
