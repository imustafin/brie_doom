	sdl_create_window (title: STRING_8; x: INTEGER_32; y: INTEGER_32; w: INTEGER_32; h: INTEGER_32; flags: NATURAL_32): detachable SDL_WINDOW_STRUCT_API
		local
			l_str: C_STRING
			l_ptr: POINTER
		do
			create l_str.make (title)
			l_ptr := sdl_create_window_api (l_str.item, x, y, w, h, flags)
			if l_ptr /= default_pointer then
				create Result.make_by_pointer (l_ptr)
			end
		ensure
			intance_free: class
		end
