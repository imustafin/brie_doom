class 
	TEST_NUMBER_CONVERSIONS

inherit
	EQA_TEST_SET

create 
	default_create

feature 

	neg_int_to_natural
		local
			int: INTEGER_32
			nat: NATURAL_32
		do
			int := -1
			nat := int.to_natural_32
			assert ("same bits", int.to_hex_string ~ nat.to_hex_string)
		end

	neg_int_as_natural
		local
			int: INTEGER_32
			nat: NATURAL_32
		do
			int := -1
			nat := int.as_natural_32
			print (int.to_hex_string + "%N")
			print (nat.to_hex_string + "%N")
			assert ("same bits", int.to_hex_string ~ nat.to_hex_string)
		end
	
end -- class TEST_NUMBER_CONVERSIONS
