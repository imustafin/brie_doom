note
	description: "m_menu.c"

class 
	MENUITEM_T

create 
	make

feature 

	status: INTEGER_32

	name: STRING_8

	routine: PROCEDURE [INTEGER_32] assign set_routine

	alphakey: CHARACTER_8
	
feature 

	make (a_status: like status; a_name: like name; a_routine: like routine; a_alphakey: like alphakey)
		do
			status := a_status
			name := a_name
			routine := a_routine
			alphakey := a_alphakey
		end
	
feature 

	set_routine (a_routine: like routine)
		do
			routine := a_routine
		end
	
end -- class MENUITEM_T
