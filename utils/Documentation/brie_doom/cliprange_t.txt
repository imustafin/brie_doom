note
	description: "[
		cliprange_t from r_bsp.c
		
		ClipWallSegment
		Clips the given range of columns
		and includes it in the new clip list.
	]"

class 
	CLIPRANGE_T

inherit
	ANY
		redefine
			out
		end

create 
	default_create

feature 

	first: INTEGER_32 assign set_first

	set_first (a_first: like first)
		do
			first := a_first
		end

	last: INTEGER_32 assign set_last

	set_last (a_last: INTEGER_32)
		do
			last := a_last
		end
	
feature 

	out: STRING_8
			-- New string containing terse printable representation
			-- of current object
		do
			Result := "(" + first.out + ", " + last.out + ")"
		end
	
end -- class CLIPRANGE_T
