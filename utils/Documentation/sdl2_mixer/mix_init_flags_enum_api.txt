class 
	MIX_INIT_FLAGS_ENUM_API

create 
	default_create

feature 

	is_valid_enum (a_value: INTEGER_32): BOOLEAN
			-- Is `a_value` a valid integer code for this enum ?
		do
			Result := a_value = mix_init_flac or a_value = mix_init_mod or a_value = mix_init_mp3 or a_value = mix_init_ogg or a_value = mix_init_mid
		end

	mix_init_flac: INTEGER_32
		external
			"C inline use <SDL_mixer.h>"
		alias
			"MIX_INIT_FLAC"
		end

	mix_init_mod: INTEGER_32
		external
			"C inline use <SDL_mixer.h>"
		alias
			"MIX_INIT_MOD"
		end

	mix_init_mp3: INTEGER_32

			"MIX_INIT_MP3"
		end

	mix_init_ogg: INTEGER_32
		external
			"C inline use <SDL_mixer.h>"
		alias
			"MIX_INIT_OGG"
		end

	mix_init_mid: INTEGER_32
		external
			"C inline use <SDL_mixer.h>"
		alias
			"MIX_INIT_MID"
		end
	
end -- class MIX_INIT_FLAGS_ENUM_API
