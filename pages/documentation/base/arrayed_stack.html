feature {NONE} -- Creation

	make_from_iterable (other: ITERABLE [G])
			-- Create a stack with all items obtained from other.
		do
			make (estimated_count_of (other))
			across
				other as o
			loop
				extend (o.item)
			end
		end
	
feature -- Element change

	extend (v: like item)
			-- Push v on top.
			-- Was declared in ARRAYED_STACK as synonym of put and force.
		do
			al_extend (v)
			start
		end

	put (v: like item)
			-- Push v on top.
			-- Was declared in ARRAYED_STACK as synonym of extend and force.
		do
			al_extend (v)
			start
		end

	force (v: like item)
			-- Push v on top.
			-- Was declared in ARRAYED_STACK as synonym of extend and put.
		do
			al_extend (v)
			start
		end
	
feature -- Removal

	remove
			-- Remove top item.
		require else
			not_empty: count /= 0
		do
			al_remove
			start
		end
	
feature -- Conversion

	linear_representation: ARRAYED_LIST [G]
			-- Representation as a linear structure
			-- (in the reverse order of original insertion)
		local
			i: INTEGER_32
		do
			from
				create Result.make (count)
				i := count
			until
				i < 1
			loop
				Result.extend (i_th (i))
				i := i - 1
			end
		end
