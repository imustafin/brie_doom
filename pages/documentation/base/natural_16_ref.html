		end

				Result := 1

feature -- Status report

	divisible (other: like Current): BOOLEAN
			-- May current object be divided by other?
		do
			Result := other.item /= 0
		ensure then
			value: Result = (other.item /= 0)
		end

	exponentiable (other: NUMERIC): BOOLEAN
			-- May current object be elevated to the power other?
		do
			if attached {INTEGER_32_REF} other as integer_value then
				Result := integer_value.item >= 0 or item /= 0
			elseif attached {REAL_32_REF} other as real_value then
				Result := real_value.item >= 0.0 or item /= 0
			elseif attached {REAL_64_REF} other as double_value then
				Result := double_value.item >= 0.0 or item /= 0
			end
		ensure then
			safe_values: ((other.conforms_to (0) and item /= 0) or (other.conforms_to (0.0) and item > 0)) implies Result
		end

	is_hashable: BOOLEAN
			-- May current object be hashed?
			-- (True if it is not its type's default.)
		do
			Result := item /= 0
		end

	is_valid_character_code: BOOLEAN
		obsolete "Use `is_valid_character_8_code' instead. [2017-05-31]"
			-- Does current object represent a CHARACTER_8?
		do
			Result := is_valid_character_8_code
		end

	is_valid_character_8_code: BOOLEAN
			-- Does current object represent a CHARACTER_8?
		do
			Result := item.to_integer_32 <= {CHARACTER_8}.max_value
		ensure
			in_bounds: Result = (item.to_integer_32 >= {CHARACTER_8}.min_value and item.to_integer_32 <= {CHARACTER_8}.max_value)
		end

	is_valid_character_32_code: BOOLEAN
			-- Does current object represent a CHARACTER_32?
		do
			Result := True
		ensure
			in_bounds: Result = (item.to_natural_32 >= {CHARACTER_32}.min_value and item.to_natural_32 <= {CHARACTER_32}.max_value)
		end

	to_hex_string: STRING_8
			-- item converted into a hexadecimal string.
		local
			i, val: INTEGER_32
		do
			from
				i := {PLATFORM}.natural_16_bits // 4
				create Result.make_filled ('0', i)
				val := item.to_integer_32
			until
				i = 0
			loop
				Result.put ((val & 15).to_hex_character, i)
				val := val |>> 4
				i := i - 1
			end
		ensure
			result_not_void: Result /= Void
			result_valid_count: Result.count = {PLATFORM}.integer_16_bits // 4
		end

	to_hex_character: CHARACTER_8
			-- item converted into a hexadecimal character.
		require
			in_bounds: 0 <= item.to_integer_32 and item <= 15
		local
			i: INTEGER_32
		do
			i := item.to_integer_32
			Result := if i <= 9 then
				'0'
			else
				'A' - 10
			end + i
		ensure
			valid_character: ("0123456789ABCDEF").has (Result)
		end

	frozen bit_shift (n: INTEGER_32): NATURAL_16
			-- Current shifted from n position to right if n positive,
			-- to left otherwise.
		require

			-- Test n-th position of Current.

		do

	frozen set_bit (b: BOOLEAN; n: INTEGER_32): NATURAL_16
			-- Copy of current with n-th position
			-- set to 1 if b, 0 otherwise.
		require
			n_nonnegative: n >= 0
			n_less_than_16: n < 16
		do
			if b then
				Result := item | ((1).to_natural_16 |<< n)
			else
				Result := item & ((1).to_natural_16 |<< n).bit_not
			end
		end

	frozen set_bit_with_mask (b: BOOLEAN; m: NATURAL_16): NATURAL_16
			-- Copy of current with all 1 bits of m set to 1
			-- if b, 0 otherwise.
		do
			if b then
				Result := item | m
			else
				Result := item & m.bit_not
			end
		end
