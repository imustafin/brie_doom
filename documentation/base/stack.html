deferred class 
	STACK [G]

inherit
	DISPENSER [G]
		export
			{NONE} prune, prune_all
		redefine
			extend,
			force,
			put,
			fill
		end

feature -- Element change

	extend (v: like item)
			-- Push v onto top.
			-- Was declared in STACK as synonym of force and put.
		deferred
		ensure then
			item_pushed: item = v
		end

	force (v: like item)
			-- Push v onto top.
			-- Was declared in STACK as synonym of extend and put.
		deferred
		ensure then
			item_pushed: item = v
		end

	put (v: like item)
			-- Push v onto top.
			-- Was declared in STACK as synonym of extend and force.
		deferred
		ensure then
			item_pushed: item = v
		end

	replace (v: like item)
			-- Replace top item by v.
		do
			remove
			extend (v)
		end

	fill (other: LINEAR [G])
			-- Fill with as many items of other as possible.
			-- Fill items with greatest index from other first.
			-- Items inserted with lowest index (from other) will
			-- always be on the top of stack.
			-- The representations of other and current structure
			-- need not be the same.
		local
			temp: ARRAYED_STACK [G]
		do
			create temp.make (0)
			from
				other.start
			until
				other.off
			loop
				temp.extend (other.item)
				other.forth
			end
			from
			until
				temp.is_empty or else not extendible
			loop
				extend (temp.item)
				temp.remove
			end
		end
