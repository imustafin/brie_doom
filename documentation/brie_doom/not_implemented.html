class 
	NOT_IMPLEMENTED

create 
	default_create

feature 

	Log_non_fatals: BOOLEAN = False

	not_implemented (name: STRING_8; fatal: BOOLEAN)
			-- Log name if log_level is high enough.
			-- If fatal, raise exception
		do
			if Log_non_fatals or fatal then
				print (name + " is not implemented%N")
			end
			if fatal then
				{I_MAIN}.i_error (name + " is not implemented%N")
			end
		ensure
			instance_free: class
				fatal implies False
		end
	
end -- class NOT_IMPLEMENTED
